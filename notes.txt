createYoutubePlayer(frame, videoId) 
{
    var div, playerId;

    playerId = 'youtube-' + videoId;
    div = document.createElement('div');
    div.setAttribute('id', playerId);

    frame.append(div);

    return new YT.Player(playerId, {
        width: '640',
        height: '360',
        videoId: videoId,
        autoplay: 1
    });
}
